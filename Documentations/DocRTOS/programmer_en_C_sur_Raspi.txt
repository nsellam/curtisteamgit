Gestion des I/O sur Raspberry en langage C

#include <stdio.h>
int main(void) {
  FILE *pFile;
  pFile = fopen("/sys/class/gpio/gpio4/value", "w");
  if ( pFile != NULL ) {
    fputs("1", pFile);
    fclose(pFile);
  }
  return 0;
}

#include <wiringPi.h>
int main(void) {
  if ( wiringPiSetup () == -1 )  
    return 1 ;
  pinMode(4, OUTPUT);
  digitalWrite(4, 1) ;
  return 0;
}
